654 template<typename _Tp>
655 void* _any_caster(const any* a)
656 {
657  // any_cast<T> returns non-null if __any->type() == typeid(T) and
658  // typeid(T) ignores cv-qualifiers so remove them:
659  using _Up = typename std::remove_cv<_Tp>::type;
660  // The contained value has a decayed type, so if decay_t<U> is not U,
661  // then it's not possible to have a contained value of type U:
662  if (!std::is_same<typename std::decay<_Up>::type, _Up>::value) {
663  return nullptr;
664  }
665  // Only copy constructible types can be used for contained values:
666  else if (!std::is_copy_constructible<_Up>::value) {
667  return nullptr;
668  }
669  // First try comparing function addresses, which works without RTTI
670  else if (a->m_manager == &any::Manager<_Up>::S_manage
671 #if __cpp_rtti
672  || a->type() == typeid(_Tp)
673 #endif
674  )
675  {
676  any::ManageArg marg;
677  a->m_manager(any::MOAccess, a, &marg);
678  return marg._obj;
679  }
680  return nullptr;
681 }

719 template<typename _Tp>
720 void any::Manager_internal<_Tp>::S_manage(ManageOption which, const any* a, ManageArg* arg)
721 {
722  // The contained object is in _M_storage._M_buffer
723  auto ptr = reinterpret_cast<const _Tp*>(&a->m_storage.m_buffer);
724  switch (which)
725  {
726  case MOAccess:
727  arg->_obj = const_cast<_Tp*>(ptr);
728  break;
729  case MOGetTypeInfo:
730 #if __cpp_rtti
731  arg->_typeinfo = &typeid(_Tp);
732 #endif
733  break;
734  case MOClone:
735  ::new(&arg->_any->m_storage.m_buffer) _Tp(*ptr);
736  arg->_any->m_manager = a->m_manager;
737  break;
738  case MODestroy:
739  ptr->~_Tp();
740  break;
741  case MOXfer:
742  ::new(&arg->_any->m_storage.m_buffer) _Tp
743  (std::move(*const_cast<_Tp*>(ptr)));
744  ptr->~_Tp();
745  arg->_any->m_manager = a->m_manager;
746  const_cast<any*>(a)->m_manager = nullptr;
747  break;
748  }
749 }
