538 template<typename _Tp>
539 void* _any_caster(const any* a)
540 {
541  // any_cast<T> returns non-null if __any->type() == typeid(T) and
542  // typeid(T) ignores cv-qualifiers so remove them:
543  using _Up = typename std::remove_cv<_Tp>::type;
544  // The contained value has a decayed type, so if decay_t<U> is not U,
545  // then it's not possible to have a contained value of type U:
546  if (!std::is_same<typename std::decay<_Up>::type, _Up>::value) {
547  return nullptr;
548  }
549  // Only copy constructible types can be used for contained values:
550  else if (!std::is_copy_constructible<_Up>::value) {
551  return nullptr;
552  }
553  // First try comparing function addresses, which works without RTTI
554  else if (a->m_manager == &any::Manager<_Up>::S_manage
555 #if __cpp_rtti
556  || a->type() == typeid(_Tp)
557 #endif
558  )
559  {
560  any::ManageArg marg;
561  a->m_manager(any::MOAccess, a, &marg);
562  return marg._obj;
563  }
564  return nullptr;
565 }

604 template<typename _Tp>
605 void any::Manager_internal<_Tp>::S_manage(ManageOption which, const any* a, ManageArg* arg)
606 {
607  // The contained object is in _M_storage._M_buffer
608  auto ptr = reinterpret_cast<const _Tp*>(&a->m_storage.m_buffer);
609  switch (which)
610  {
611  case MOAccess:
612  arg->_obj = const_cast<_Tp*>(ptr);
613  break;
614  case MOGetTypeInfo:
615 #if __cpp_rtti
616  arg->_typeinfo = &typeid(_Tp);
617 #endif
618  break;
619  case MOClone:
620  ::new(&arg->_any->m_storage.m_buffer) _Tp(*ptr);
621  arg->_any->m_manager = a->m_manager;
622  break;
623  case MODestroy:
624  ptr->~_Tp();
625  break;
626  case MOXfer:
627  ::new(&arg->_any->m_storage.m_buffer) _Tp
628  (std::move(*const_cast<_Tp*>(ptr)));
629  ptr->~_Tp();
630  arg->_any->m_manager = a->m_manager;
631  const_cast<any*>(a)->m_manager = nullptr;
632  break;
633  }
634 }
